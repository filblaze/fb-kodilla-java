SELECT
    IL.NAME as ISSUELIST,
    I.ID as IssueID,
    I.SUMMARY,
    I.DESCRIPTION
From issues I
JOIN issueslists IL ON I.ISSUESLIST_ID = IL.ID;

//2
SELECT
    CONCAT(U.FIRSTNAME,' ', U.LASTNAME) as NAME,
    I.ID as IssueID,
    I.SUMMARY,
    I.DESCRIPTION
FROM issues I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

//3
SELECT
    CONCAT(U.FIRSTNAME, ' ', U.LASTNAME) as NAME,
    COUNT(I.ID) AS LICZBA_ZADAN
FROM users U
JOIN issues I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY U.ID, NAME;
